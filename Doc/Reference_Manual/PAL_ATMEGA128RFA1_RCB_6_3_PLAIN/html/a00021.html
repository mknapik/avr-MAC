00470 retval_t pal_timer_stop(uint8_t timer_id)
00471 {
00472     bool timer_stop_request_status = false;
00473     uint8_t curr_index;
00474     uint8_t prev_index;
00475 
00476 
00477     if (timer_id >= TOTAL_NUMBER_OF_TIMERS)
00478     {
00479         return (PAL_TMR_INVALID_ID);
00480     }
00481 
00482     ENTER_CRITICAL_REGION();
00483 
00484     /* Check if any timer has expired. */
00485     internal_timer_handler();
00486 
00487     /* The requested timer is first searched in the running timer queue */
00488     if (running_timers > 0)
00489     {
00490         uint8_t timer_count = running_timers;
00491         prev_index = curr_index = running_timer_queue_head;
00492         while (timer_count > 0)
00493         {
00494             if (timer_id == curr_index)
00495             {
00496                 timer_stop_request_status = true;
00497 
00498                 if (timer_id == running_timer_queue_head)
00499                 {
00500                     running_timer_queue_head =
00501                         timer_array[timer_id].next_timer_in_queue;
00502                     /*
00503                      * The value in OCR corresponds to the timeout pointed
00504                      * by the 'running_timer_queue_head'. As the head has
00505                      * changed here, OCR needs to be loaded by the new
00506                      * timeout value, if any.
00507                      */
00508                     prog_ocr();
00509                 }
00510                 else
00511                 {
00512                     timer_array[prev_index].next_timer_in_queue =
00513                         timer_array[timer_id].next_timer_in_queue;
00514                 }
00515                 /*
00516                  * The next timer element of the stopped timer is updated
00517                  * to its default value.
00518                  */
00519                 timer_array[timer_id].next_timer_in_queue = NO_TIMER;
00520                 break;
00521             }
00522             else
00523             {
00524                 prev_index = curr_index;
00525                 curr_index = timer_array[curr_index].next_timer_in_queue;
00526             }
00527             timer_count--;
00528         }
00529         if (timer_stop_request_status)
00530         {
00531             running_timers--;
00532         }
00533     }
00534 
00535     /*
00536      * The requested timer is not present in the running timer queue.
00537      * It will be now searched in the expired timer queue
00538      */
00539     if (!timer_stop_request_status)
00540     {
00541         prev_index = curr_index = expired_timer_queue_head;
00542         while (NO_TIMER != curr_index)
00543         {
00544             if (timer_id == curr_index)
00545             {
00546                 if (timer_id == expired_timer_queue_head)
00547                 {
00548                     /*
00549                      * The requested timer is the head of the expired timer
00550                      * queue
00551                      */
00552                     if (expired_timer_queue_head == expired_timer_queue_tail)
00553                     {
00554                         /* Only one timer in expired timer queue */
00555                         expired_timer_queue_head = expired_timer_queue_tail =
00556                             NO_TIMER;
00557                     }
00558                     else
00559                     {
00560                         /*
00561                          * The head of the expired timer queue is moved to next
00562                          * timer in the expired timer queue.
00563                          */
00564                         expired_timer_queue_head =
00565                             timer_array[expired_timer_queue_head].next_timer_in_queue;
00566                     }
00567                 }
00568                 else
00569                 {
00570                     /*
00571                      * The requested timer is present in the middle or at the
00572                      * end of the expired timer queue.
00573                      */
00574                     timer_array[prev_index].next_timer_in_queue =
00575                         timer_array[timer_id].next_timer_in_queue;
00576 
00577                     /*
00578                      * If the stopped timer is the one which is at the tail of
00579                      * the expired timer queue, then the tail is updated.
00580                      */
00581                     if (timer_id == expired_timer_queue_tail)
00582                     {
00583                         expired_timer_queue_tail = prev_index;
00584                     }
00585                 }
00586                 timer_stop_request_status = true;
00587                 break;
00588             }
00589             else
00590             {
00591                prev_index = curr_index;
00592                curr_index = timer_array[curr_index].next_timer_in_queue;
00593             }
00594         }
00595     }
00596 
00597     if (timer_stop_request_status)
00598     {
00599         /*
00600          * The requested timer is stopped, hence the structure elements of the
00601          * timer are updated.
00602          */
00603         timer_array[timer_id].timer_cb = NULL;
00604     }
00605 
00606     LEAVE_CRITICAL_REGION();
00607 
00608     if (timer_stop_request_status)
00609     {
00610         return (MAC_SUCCESS);
00611     }
00612 
00613     return (PAL_TMR_NOT_RUNNING);
00614 }

00662 void timer_init(void)
00663 {
00664 #if ((TOTAL_NUMBER_OF_TIMERS > 0) || defined(DOXYGEN))
00665     /*
00666      * Initialize the timer resources like timer arrays
00667      * queues, timer registers
00668      */
00669     uint8_t index;
00670 
00671     running_timers = 0;
00672     timer_trigger = false;
00673 
00674     running_timer_queue_head = NO_TIMER;
00675     expired_timer_queue_head = NO_TIMER;
00676     expired_timer_queue_tail = NO_TIMER;
00677 #ifdef ENABLE_HIGH_PRIO_TMR
00678     high_priority_timer_id = NO_TIMER;
00679 #endif
00680 
00681     for (index = 0; index < TOTAL_NUMBER_OF_TIMERS; index++)
00682     {
00683         timer_array[index].next_timer_in_queue = NO_TIMER;
00684         timer_array[index].timer_cb = NULL;
00685     }
00686 #endif  /* #if ((TOTAL_NUMBER_OF_TIMERS > 0) || defined(DOXYGEN)) */
00687 
00688     /*
00689      * Enable the 32 kHz crystal oscillator. This is required in
00690      * order to allow for the hardware performing a transparent clock
00691      * switch when the main 16 MHz crystal is turned off (i.e., when
00692      * the transceiver module is put to sleep).
00693      */
00694     ASSR = _BV(AS2);
00695 
00696     /*
00697      * Start the MAC symbol counter, resetting the prescaler by the
00698      * same time.  Also enable the rx timestamping feature.
00699      */
00700     SCCR0 = _BV(SCTSE) | _BV(SCEN) | _BV(SCRES);
00701 
00702     /*
00703      * Use SCOCR2 to implement a virtual 28-bit timer. This is needed
00704      * since all layers above PAL operate at a microsecond level, while
00705      * the MAC symbol counter operates at a granularity of MAC symbol
00706      * periods (16 s).  Due to that, the timer values have to be
00707      * scaled at the PAL API border, and this scaling effectively
00708      * reduces the timer span to 28 bits.
00709      *
00710      * Whenever the timer is about to cross the 28-bit boundary, it is
00711      * advanced (by the ISR for SCOCR2) so the hardware will then
00712      * perform a real rollover at the next timer tick.  That way, the
00713      * worst effect from blocking interrupts for too long is that all
00714      * pending timers at the time of rollover will be delayed a bit.
00715      * If the SCOCR2 interrupt is served within 16 s, no additional
00716      * delay will happen.
00717      */
00718     SC_WRITE32(SCOCR2, 0x0FFFFFFFul);
00719     SCIRQS = _BV(IRQSCP2);
00720     SCIRQM |= _BV(IRQMCP2);
00721 }

00786 void  pal_timer_delay(uint16_t delay)
00787 {
00788     bool needs_cli = false;
00789     uint8_t sreg = 0;
00790 
00791     /*
00792      * Any interrupt occurring during the delay calculation will introduce
00793      * additional delay and can also affect the logic of delay calculation.
00794      * Hence the delay implementation is put under critical region as long
00795      * as only a small delay time is requested.
00796      */
00797     if (delay < DELAY_WITHOUT_IRQS_US)
00798     {
00799         needs_cli = true;
00800         sreg = SREG;
00801         cli();
00802     }
00803 
00804     /*
00805      * The logic below depends on the CPU being clocked with 16 MHz,
00806      * and timer 5 being set up for a prescaler of 8, so it will be
00807      * clocked with 2 MHz.  This is the best compromise to achieve the
00808      * 1 s resolutions callers of pal_timer_delay() are expecting.
00809      *
00810      * In case any other CPU clock frequency should be supported that
00811      * cannot work this way, some of the logic below needs to be
00812      * rewritten.  Normally, there should be no real need for a
00813      * different CPU clock frequency, as the ATmega128RFA1 can operate
00814      * at 16 MHz down to the minimal possible supply voltage.
00815      *
00816      * As a special exception, the CPU frequency 15.3846 MHz will be
00817      * tolerated as it is used in some test setups.  This results in
00818      * delays being approximately 4 % too long which is deemed to be
00819      * acceptable for that particular purpose.
00820      */
00821 #if (F_CPU != 16000000UL) && (F_CPU != 15384600UL)
00822 #  error "pal_timer_delay() is only implemented for F_CPU = 16 MHz"
00823 #endif
00824     /*
00825      * With a 2 MHz timer clock, it is not possible to span the entire
00826      * delay range of 65 ms using a single timer 5 cycle, so the delay
00827      * is split into multiple smaller delays if required.
00828      */
00829     while (delay != 0)
00830     {
00831         uint16_t actval;
00832 
00833         /*
00834          * Find whether the entire delay time will fit or not.  Note
00835          * that the 2 MHz figure below must *not* be calculated as
00836          * F_CPU / (1000000 * 8) because that will not work as
00837          * expected in the the 15.3 MHz case.
00838          */
00839         if (delay > UINT16_MAX / 2 /* MHz timer clock */)
00840         {
00841             actval = UINT16_MAX / 2;
00842         }
00843         else
00844         {
00845             actval = delay;
00846         }
00847         delay -= actval;        /* remaining delay for next round */
00848 
00849         /*
00850          * Perform the actual delay.  Setup TCNT5 so it will overflow
00851          * after actval cycles.  Then, clear any pending overflow
00852          * interrupt, and start timer 5 with a prescaler of 8.  Poll
00853          * for the overflow interrupt flag, and stop the timer again
00854          * afterwards.
00855          */
00856         actval *= 2 /* microseconds to timer 5 clock cycles @ 2 MHz */;
00857         TCNT5 = (uint16_t)(-actval);
00858         TIFR5 = _BV(TOV5);
00859         TCCR5B = _BV(CS51);
00860         while ((TIFR5 & _BV(TOV5)) == 0)
00861             /* delay */;
00862         TCCR5B = 0;
00863     }
00864 
00865     if (needs_cli)
00866     {
00867         SREG = sreg;
00868     }

00931 static void prog_ocr(void)
00932 {
00933     uint32_t new_ocr = timer_array[running_timer_queue_head].abs_exp_timer;
00934 
00935     if (running_timers != 0)
00936     {
00937         SC_WRITE32(SCOCR1, new_ocr);
00938         if ((SCIRQM & _BV(IRQMCP1)) == 0)
00939         {
00940             SCIRQS = _BV(IRQSCP1);
00941             SCIRQM |= _BV(IRQMCP1);
00942         }
00943 
00944         /* Safety check: Trigger timer, if next_trigger is in the past. */
00945         if (compare_time(new_ocr, SC_READ32(SCCNT) + 1))
00946         {
00947             timer_trigger = true;

00949         else if ((VERSION_NUM <= 7) && (timer_last_trigger + 1 == new_ocr))

00955              *
00956              * Just wait a little moment here, and then proceed as if
00957              * the ISR had just triggered.
00958              */

00960                 /* wait */;
00961             timer_trigger = true;

00966         SCIRQM &= ~_BV(IRQMCP1);

01045 static void start_absolute_timer(uint8_t timer_id,
01046                           uint32_t point_in_time,
01047                           FUNC_PTR handler_cb,
01048                           void *parameter)
01049 {
01050     ENTER_CRITICAL_REGION();
01051 
01052     point_in_time &= MASK_LOW28;
01053 
01054     /* Check is done to see if any timer has expired */
01055     internal_timer_handler();
01056 
01057     if (NO_TIMER == running_timer_queue_head)
01058     {
01059         running_timer_queue_head = timer_id;
01060         timer_array[timer_id].next_timer_in_queue = NO_TIMER;
01061     }
01062     else
01063     {
01064         uint8_t i;
01065         bool timer_inserted = false;
01066         uint8_t curr_index = running_timer_queue_head;
01067         uint8_t prev_index = running_timer_queue_head;
01068 
01069         for (i = 0; i < running_timers; i++)
01070         {
01071             if (NO_TIMER != curr_index)
01072             {
01073                 if (compare_time(timer_array[curr_index].abs_exp_timer,
01074                                      point_in_time))
01075                 {
01076                     /*
01077                      * Requested absolute time value is greater than the time
01078                      * value pointed by the curr_index in the timer array
01079                      */
01080                     prev_index = curr_index;
01081                     curr_index = timer_array[curr_index].next_timer_in_queue;
01082                 }
01083                 else
01084                 {
01085                     timer_array[timer_id].next_timer_in_queue = curr_index;
01086                     if (running_timer_queue_head == curr_index)
01087                     {
01088                         /* Insertion at the head of the timer queue. */
01089                         running_timer_queue_head = timer_id;
01090                     }
01091                     else
01092                     {
01093                         timer_array[prev_index].next_timer_in_queue = timer_id;
01094                     }
01095                     timer_inserted = true;
01096                     break;
01097                 }
01098             }
01099         }
01100         if (!timer_inserted)
01101         {
01102             /* Insertion at the tail of the timer queue. */
01103             timer_array[prev_index].next_timer_in_queue = timer_id;
01104             timer_array[timer_id].next_timer_in_queue = NO_TIMER;
01105         }
01106     }
01107     timer_array[timer_id].abs_exp_timer = point_in_time;
01108     timer_array[timer_id].timer_cb = (FUNC_PTR)handler_cb;
01109     timer_array[timer_id].param_cb = parameter;
01110     running_timers++;
01111 
01112     prog_ocr();
01113 
01114     LEAVE_CRITICAL_REGION();
01115 }
