00051 void pal_trx_read_timestamp(uint32_t *timestamp)
00052 {
00053     /*
00054      * Everytime a transceiver interrupt is triggred, input capture register of
00055      * the AVR is latched. The 'sys_time' is concatenated to the ICR to
00056      * generate the time stamp of the received frame.
00057      * 'sys_time'   'ICR'
00058      *  ---------|--------- => 32 bit timestamp
00059      *   16 bits   16 bits
00060      */
00061     *timestamp = (uint32_t)sys_time << (uint32_t)16;
00062     *timestamp |= (uint32_t)TIME_STAMP_REGISTER;
00063 }

00075 bool pal_calibrate_rc_osc(void)
00076 {
00077     /*
00078      * Use the 32.768 kHz external crystal oscillator as reference clock source.
00079      */
00080     /*
00081      * This is the actual value of the timer to be calibrated
00082      * after each calibration run.
00083      */
00084     uint16_t curr_value_cal_timer = 0;
00085 
00086     /* This is the best OSCCAL value. */
00087     uint8_t best_value_osccal = 0;
00088     /*
00089      * This is the smallest difference between the target value and the actual value
00090      * of the timer the timer to be calibrated.
00091      */
00092 
00093     uint16_t best_dif_timers = ~(uint16_t)0;
00094     /*
00095      * This is the difference between  between the target value and the actual value
00096      * of the timer the timer to be calibrated of the current calibration attempt.
00097      */
00098     uint16_t curr_dif_timers = 0;
00099 
00100     uint16_t counter;
00101     uint8_t tccr2b;
00102     uint8_t tccr1b;
00103     uint8_t tccr1a;
00104     bool cal_result = true;
00105 
00106     ENTER_CRITICAL_REGION();
00107 
00108     /*
00109      * Save current values of timer status.
00110      */
00111     tccr2b = TCCR2B;
00112     tccr1b = TCCR1B;
00113     tccr1a = TCCR1A;
00114 
00115     /*
00116      * Stop timers 1 and 2.
00117      * Set timer 1 to normal mode (no CTC, no PWM, just count).
00118      */
00119     TCCR2B = 0;
00120     TCCR1B = 0;
00121     TCCR1A = 0;
00122 
00123     for (counter = 0; counter < MAX_CAL_LOOP_CNT;  counter++)
00124     {
00125         /*
00126          * Delete pending timer 1 and 2 interrupts, and clear the
00127          * counters.
00128          */
00129         TIFR1 = 0xFF;
00130         TIFR2 = 0xFF;
00131         TCNT2 = 0;
00132         TCNT1 = 0;
00133 
00134         /* Kick timer1 with internal clock source and no prescaler */
00135         TCCR1B = (1 << CS10);
00136 
00137         /*
00138          * Kick timer2 with external 32.768 Hz asynchronous clock
00139          * and no prescaler
00140          */
00141         TCCR2B = (1 << CS20);
00142         ASSR = (1 << AS2);
00143 
00144         /*
00145          * Wait for timer 2 to overflow.
00146          */
00147         while (!(TIFR2 & (1 << TOV2)))
00148         {
00149             /* Wait */
00150         }
00151 
00152 
00153         /*
00154          * Stop timer 1.  Now, TCNT1 contains the number of CPU cycles
00155          * counted during F_CPU / (32 * 256) cycles.
00156          */
00157         TCCR1B = 0;
00158         TCCR2B = 0;
00159 
00160         curr_value_cal_timer = TCNT1;
00161 
00162         /*
00163          * Check if reference timer is running at all,
00164          * i.e. the reference clock is available.
00165          */
00166         if (0 == curr_value_cal_timer)
00167         {
00168             /* Reference timer is not running, return error. */
00169             cal_result = false;
00170             break;
00171         }
00172 
00173         if (curr_value_cal_timer <= (uint16_t)(TARGETVAL_CALIBRATION))
00174         {
00175             curr_dif_timers = (uint16_t)(TARGETVAL_CALIBRATION) - curr_value_cal_timer;
00176         }
00177         else
00178         {
00179             curr_dif_timers = curr_value_cal_timer - (uint16_t)(TARGETVAL_CALIBRATION);
00180         }
00181 
00182         if (curr_dif_timers < best_dif_timers)
00183         {
00184             best_dif_timers = curr_dif_timers;
00185             best_value_osccal = OSCCAL;
00186         }
00187 
00188         if (curr_value_cal_timer <= (uint16_t)(TARGETVAL_CALIBRATION))
00189         {
00190             /* Too slow, increase speed */
00191             OSCCAL++;
00192         }
00193         else
00194         {
00195             /* Too fast, lower speed */
00196             OSCCAL--;
00197         }
00198     }
00199 
00200     TCCR2B = tccr2b;
00201     TCCR1B = tccr1b;
00202     TCCR1A = tccr1a;
00203     LEAVE_CRITICAL_REGION();
00204 
00205     OSCCAL = best_value_osccal;
00206 
00207     return (cal_result);
00208 }
