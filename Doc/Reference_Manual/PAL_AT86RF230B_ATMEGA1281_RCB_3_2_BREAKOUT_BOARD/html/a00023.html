00053 {
00054     /* Calculate corresponding value for baud rateregister. */
00055     uint16_t baud_rate_reg = UART_BAUD(baud_rate);
00056 
00057     /*
00058      * Microcontroller's USART register is updated to
00059      * run at the given baud rate.
00060      */
00061     UBRR0H = (baud_rate_reg >> 8) & 0xFF;
00062     UBRR0L = (uint8_t)baud_rate_reg;

00331 
00332     if (UART_MAX_RX_BUF_LENGTH <= uart_0_buffer.rx_count)
00333     {
00334         /*
00335          * Bytes between head and tail are overwritten by new data.
00336          * The oldest data in buffer is the one to which the tail is
00337          * pointing. So reading operation should start from the tail.
00338          */

00340

00350             /*

00356         }
00357     }
