00276 retval_t pal_timer_stop(uint8_t timer_id)
00277 {
00278     bool timer_stop_request_status = false;
00279     uint8_t curr_index;
00280     uint8_t prev_index;
00281 
00282 
00283     if (timer_id >= TOTAL_NUMBER_OF_TIMERS)
00284     {
00285         return (PAL_TMR_INVALID_ID);
00286     }
00287 
00288     ENTER_CRITICAL_REGION();
00289 
00290     /* Check if any timer has expired. */
00291     internal_timer_handler();
00292 
00293     /* The requested timer is first searched in the running timer queue */
00294     if (running_timers > 0)
00295     {
00296         uint8_t timer_count = running_timers;
00297         prev_index = curr_index = running_timer_queue_head;
00298         while (timer_count > 0)
00299         {
00300             if (timer_id == curr_index)
00301             {
00302                 timer_stop_request_status = true;
00303 
00304                 if (timer_id == running_timer_queue_head)
00305                 {
00306                     running_timer_queue_head =
00307                         timer_array[timer_id].next_timer_in_queue;
00308                     /*
00309                      * The value in OCR corresponds to the timeout pointed
00310                      * by the 'running_timer_queue_head'. As the head has
00311                      * changed here, OCR needs to be loaded by the new
00312                      * timeout value, if any.
00313                      */
00314                     prog_ocr();
00315                 }
00316                 else
00317                 {
00318                     timer_array[prev_index].next_timer_in_queue =
00319                         timer_array[timer_id].next_timer_in_queue;
00320                 }
00321                 /*
00322                  * The next timer element of the stopped timer is updated
00323                  * to its default value.
00324                  */
00325                 timer_array[timer_id].next_timer_in_queue = NO_TIMER;
00326                 break;
00327             }
00328             else
00329             {
00330                 prev_index = curr_index;
00331                 curr_index = timer_array[curr_index].next_timer_in_queue;
00332             }
00333             timer_count--;
00334         }
00335         if (timer_stop_request_status)
00336         {
00337             running_timers--;
00338         }
00339     }
00340 
00341     /*
00342      * The requested timer is not present in the running timer queue.
00343      * It will be now searched in the expired timer queue
00344      */
00345     if (!timer_stop_request_status)
00346     {
00347         prev_index = curr_index = expired_timer_queue_head;
00348         while (NO_TIMER != curr_index)
00349         {
00350             if (timer_id == curr_index)
00351             {
00352                 if (timer_id == expired_timer_queue_head)
00353                 {
00354                     /*
00355                      * The requested timer is the head of the expired timer
00356                      * queue
00357                      */
00358                     if (expired_timer_queue_head == expired_timer_queue_tail)
00359                     {
00360                         /* Only one timer in expired timer queue */
00361                         expired_timer_queue_head = expired_timer_queue_tail =
00362                             NO_TIMER;
00363                     }
00364                     else
00365                     {
00366                         /*
00367                          * The head of the expired timer queue is moved to next
00368                          * timer in the expired timer queue.
00369                          */
00370                         expired_timer_queue_head =
00371                             timer_array[expired_timer_queue_head].next_timer_in_queue;
00372                     }
00373                 }
00374                 else
00375                 {
00376                     /*
00377                      * The requested timer is present in the middle or at the
00378                      * end of the expired timer queue.
00379                      */
00380                     timer_array[prev_index].next_timer_in_queue =
00381                         timer_array[timer_id].next_timer_in_queue;
00382 
00383                     /*
00384                      * If the stopped timer is the one which is at the tail of
00385                      * the expired timer queue, then the tail is updated.
00386                      */
00387                     if (timer_id == expired_timer_queue_tail)
00388                     {
00389                         expired_timer_queue_tail = prev_index;
00390                     }
00391                 }
00392                 timer_stop_request_status = true;
00393                 break;
00394             }
00395             else
00396             {
00397                prev_index = curr_index;
00398                curr_index = timer_array[curr_index].next_timer_in_queue;
00399             }
00400         }
00401     }
00402 
00403     if (timer_stop_request_status)
00404     {
00405         /*
00406          * The requested timer is stopped, hence the structure elements of the
00407          * timer are updated.
00408          */
00409         timer_array[timer_id].timer_cb = NULL;
00410     }
00411 
00412     LEAVE_CRITICAL_REGION();
00413 
00414     if (timer_stop_request_status)
00415     {
00416         return (MAC_SUCCESS);
00417     }
00418 
00419     return (PAL_TMR_NOT_RUNNING);
00420 }

00502 void  pal_timer_delay(uint16_t delay)
00503 {
00504     /*
00505      * Any interrupt occurring during the delay calculation will introduce
00506      * additional delay and can also affect the logic of delay calculation.
00507      * Hence the delay implementation is put under critical region.
00508      */
00509 
00510     ENTER_CRITICAL_REGION();
00511 
00512     if (delay > MIN_DELAY_VAL)
00513     {
00514         uint16_t target_time = delay + TCC0_CNT;
00515 
00516         if (target_time > TCC0_CNT)
00517         {
00518             /* Target time is within current span of 16 bit count (0xFFFF) */
00519             while (target_time > TCC0_CNT)
00520             {
00521                 /* No operation, till condition is met */
00522             }
00523         }
00524         else
00525         {
00526             /*
00527              * Target time has rolled over and has gone to next span of 16 bit
00528              * count.
00529              */
00530             while (target_time < TCC0_CNT)
00531             {
00532                 /* No operation, till condition is met */
00533             }
00534             while (target_time > TCC0_CNT)
00535             {
00536                 /* No operation, till condition is met */
00537             }
00538         }
00539     }
00540 
00541     LEAVE_CRITICAL_REGION();
00542 }

00604 static inline uint32_t gettime(void)

00607     uint32_t current_time;

00609     do
00610     {
00611         current_sys_time = sys_time;
00612         current_time = current_sys_time;
00613         current_time = current_time << 16;

00633 static void prog_ocr(void)
00634 {
00635     uint16_t  timeout_high;
00636     uint16_t  timeout_low;
00637     uint32_t  timeout;
00638 
00639     ENTER_CRITICAL_REGION();
00640 
00641     if (NO_TIMER != running_timer_queue_head)
00642     {
00643         timeout = timer_array[running_timer_queue_head].abs_exp_timer;
00644         timeout_high = (uint16_t)(timeout >> SYS_TIME_SHIFT_MASK);
00645 
00646         if (timeout_high == sys_time)
00647         {
00648             timeout_low = (uint16_t)(timeout & HW_TIME_MASK);
00649 
00650             /* Program output compare match */
00651             TCC0_CCA = timeout_low;
00652 
00653             /* Clear pending output compare matches */
00654             TCC0_INTFLAGS |= TC0_CCAIF_bm;
00655 
00656             /* Enable output compare match interrupt */
00657             TCC0_INTCTRLB |= TC_CCAINTLVL_HI_gc;
00658         }
00659         else
00660         {
00661             /* Disable output compare match interrupt. */
00662             /* Implicit casting required to avoid IAR Pa091. */

00664         }

00666         uint32_t current_time = gettime();
00667 
00668         /* Safety check: Trigger timer, if next_trigger is in the past. */
00669         if (compare_time(timer_array[running_timer_queue_head].abs_exp_timer, current_time + 1))
00670         {
00671             timer_trigger = true;
00672         }
00673     }
00674

00689 void internal_timer_handler(void)
00690 {
00691     /*
00692      * Flag was set once a timer has expired by the timer ISR or
00693      * by function prog_rc().
00694      */
00695     if (timer_trigger)
00696     {
00697         timer_trigger = false;
00698 
00699         if (running_timers > 0) /* Holds the number of running timers */
00700         {
00701             if ((expired_timer_queue_head == NO_TIMER) &&
00702                 (expired_timer_queue_tail == NO_TIMER))
00703             {
00704                 expired_timer_queue_head = expired_timer_queue_tail =
00705                                                     running_timer_queue_head;
00706             }
00707             else
00708             {
00709                 timer_array[expired_timer_queue_tail].next_timer_in_queue =
00710                                                     running_timer_queue_head;
00711 
00712                 expired_timer_queue_tail = running_timer_queue_head;
00713             }
00714 
00715             running_timer_queue_head =
00716                 timer_array[running_timer_queue_head].next_timer_in_queue;
00717 
00718             timer_array[expired_timer_queue_tail].next_timer_in_queue =
00719                 NO_TIMER;
00720 
00721             running_timers--;
00722 
00723             /*
00724              * As a timer has expired, the OCR1A is programmed (if possible)
00725              * with the new timeout value of the timer pointed by running
00726              * timer queue head
00727              */
00728             prog_ocr();
00729         }
00730     }

00749 static void start_absolute_timer(uint8_t timer_id,
00750                           uint32_t point_in_time,
00751                           FUNC_PTR handler_cb,
00752                           void *parameter)
00753 {
00754     ENTER_CRITICAL_REGION();
00755 
00756     /* Check is done to see if any timer has expired */
00757     internal_timer_handler();
00758 
00759     bool load_ocr = false;
00760 
00761     if (NO_TIMER == running_timer_queue_head)
00762     {
00763         running_timer_queue_head = timer_id;
00764         timer_array[timer_id].next_timer_in_queue = NO_TIMER;
00765         /*
00766          * This is the only timer running in the timer queue, hence load the
00767          * OCR.
00768          */
00769         load_ocr = true;
00770     }
00771     else
00772     {
00773         uint8_t i;
00774         bool timer_inserted = false;
00775         uint8_t curr_index = running_timer_queue_head;
00776         uint8_t prev_index = running_timer_queue_head;
00777 
00778         for (i = 0; i < running_timers; i++)
00779         {
00780             if (NO_TIMER != curr_index)
00781             {
00782                 if (compare_time(timer_array[curr_index].abs_exp_timer,
00783                                      point_in_time))
00784                 {
00785                     /*
00786                      * Requested absolute time value is greater than the time
00787                      * value pointed by the curr_index in the timer array
00788                      */
00789                     prev_index = curr_index;
00790                     curr_index = timer_array[curr_index].next_timer_in_queue;
00791                 }
00792                 else
00793                 {
00794                     timer_array[timer_id].next_timer_in_queue = curr_index;
00795                     if (running_timer_queue_head == curr_index)
00796                     {
00797                         /* Insertion at the head of the timer queue. */
00798                         running_timer_queue_head = timer_id;
00799                         /*
00800                          * Timer is inserted at the head of the queue, hence
00801                          * load the OCR.
00802                          */
00803                         load_ocr = true;
00804                     }
00805                     else
00806                     {
00807                         timer_array[prev_index].next_timer_in_queue = timer_id;
00808                     }
00809                     timer_inserted = true;
00810                     break;
00811                 }
00812             }
00813         }
00814         if (!timer_inserted)
00815         {
00816             /* Insertion at the tail of the timer queue. */
00817             timer_array[prev_index].next_timer_in_queue = timer_id;
00818             timer_array[timer_id].next_timer_in_queue = NO_TIMER;
00819         }
00820     }
00821     timer_array[timer_id].abs_exp_timer = point_in_time;
00822     timer_array[timer_id].timer_cb = (FUNC_PTR)handler_cb;
00823     timer_array[timer_id].param_cb = parameter;
00824     running_timers++;
00825 
00826     /*
00827      * If there is only one timer in the timer queue
00828      * the timeout should be loaded immediately
00829      */
00830     if (load_ocr)
00831     {
00832         prog_ocr();
00833     }
00834 
00835     LEAVE_CRITICAL_REGION();
00836 }
