00159 static uint8_t eeprom_read_byte(uint8_t *addr)
00160 {
00161     uint16_t addrcpy;
00162 
00163     addrcpy = (uint16_t)addr;
00164 
00165     eeprom_wait_for_NVM();
00166 
00167     NVM.ADDR2 = 0;
00168     NVM.ADDR1 = (addrcpy >> 8) & 0xFF;
00169     NVM.ADDR0 = (addrcpy) & 0xFF;
00170     NVM.CMD =  NVM_CMD_READ_EEPROM_gc;
00171 
00172     /*
00173      * The CMDEX bit must be set within 4 clock cycles after setting the
00174      * protection byte in the CCP register.
00175      */
00176     NVM_EXEC();

00213 static void eeprom_write_byte(uint8_t addr, uint8_t value)
00214 {
00215     uint16_t addrcpy;
00216 
00217     addrcpy = (uint16_t)addr;
00218 
00219     /*  Flush buffer to make sure no unintetional data is written and load
00220      *  the "Page Load" command into the command register.
00221      */
00222     eeprom_flush_buffer();
00223     NVM.CMD = NVM_CMD_LOAD_EEPROM_BUFFER_gc;
00224 
00225     /* Set address to write to. */
00226     NVM.ADDR0 = addrcpy & 0xFF;
00227     NVM.ADDR1 = (addrcpy >> 8) & 0x1F;
00228     NVM.ADDR2 = 0x00;
00229 
00230     /* Load data to write, which triggers the loading of EEPROM page buffer. */
00231     NVM.DATA0 = value;
00232 
00233     /*  Issue EEPROM Atomic Write (Erase&Write) command. Load command, write
00234      *  the protection signature and execute command.
00235      */
00236     NVM.CMD = NVM_CMD_ERASE_WRITE_EEPROM_PAGE_gc;
00237     NVM_EXEC();
00238 }
