00060 void pal_trx_read_timestamp(uint32_t *timestamp)
00061 {
00062     /*
00063      * Everytime a transceiver interrupt is triggred, input capture register of
00064      * the AVR is latched. The 'sys_time' is concatenated to the ICR to
00065      * generate the time stamp of the received frame.
00066      * 'sys_time'   'ICR'
00067      *  ---------|--------- => 32 bit timestamp
00068      *   16 bits   16 bits
00069      */
00070     *timestamp = (uint32_t)sys_time << (uint32_t)16;
00071     *timestamp |= (uint32_t)TIME_STAMP_REGISTER;
00072 }

00084 bool pal_calibrate_rc_osc(void)
00085 {
00086     /*
00087      * Use the 32.768 kHz external crystal oscillator as reference clock source.
00088      */
00089     /*
00090      * This is the actual value of the timer to be calibrated
00091      * after each calibration run.
00092      */
00093     uint16_t curr_value_cal_timer = 0;
00094 
00095     /* This is the best OSCCAL value. */
00096     uint8_t best_value_osccal = 0;
00097     /*
00098      * This is the smallest difference between the target value and the actual value
00099      * of the timer the timer to be calibrated.
00100      */
00101 
00102     uint16_t best_dif_timers = ~(uint16_t)0;
00103     /*
00104      * This is the difference between  between the target value and the actual value
00105      * of the timer the timer to be calibrated of the current calibration attempt.
00106      */
00107     uint16_t curr_dif_timers = 0;
00108 
00109     uint16_t counter;
00110     uint8_t tccr2b;
00111     uint8_t tccr1b;
00112     uint8_t tccr1a;
00113     bool cal_result = true;
00114 
00115     ENTER_CRITICAL_REGION();
00116 
00117     /*
00118      * Save current values of timer status.
00119      */
00120     tccr2b = TCCR2B;
00121     tccr1b = TCCR1B;
00122     tccr1a = TCCR1A;
00123 
00124     /*
00125      * Stop timers 1 and 2.
00126      * Set timer 1 to normal mode (no CTC, no PWM, just count).
00127      */
00128     TCCR2B = 0;
00129     TCCR1B = 0;
00130     TCCR1A = 0;
00131 
00132     for (counter = 0; counter < MAX_CAL_LOOP_CNT;  counter++)
00133     {
00134         /*
00135          * Delete pending timer 1 and 2 interrupts, and clear the
00136          * counters.
00137          */
00138         TIFR1 = 0xFF;
00139         TIFR2 = 0xFF;
00140         TCNT2 = 0;
00141         TCNT1 = 0;
00142 
00143         /* Kick timer1 with internal clock source and no prescaler */
00144         TCCR1B = (1 << CS10);
00145 
00146         /*
00147          * Kick timer2 with external 32.768 Hz asynchronous clock
00148          * and no prescaler
00149          */
00150         TCCR2B = (1 << CS20);
00151         ASSR = (1 << AS2);
00152 
00153         /*
00154          * Wait for timer 2 to overflow.
00155          */
00156         while (!(TIFR2 & (1 << TOV2)))
00157         {
00158             /* Wait */
00159         }
00160 
00161 
00162         /*
00163          * Stop timer 1.  Now, TCNT1 contains the number of CPU cycles
00164          * counted during F_CPU / (32 * 256) cycles.
00165          */
00166         TCCR1B = 0;
00167         TCCR2B = 0;
00168 
00169         curr_value_cal_timer = TCNT1;
00170 
00171         /*
00172          * Check if reference timer is running at all,
00173          * i.e. the reference clock is available.
00174          */
00175         if (0 == curr_value_cal_timer)
00176         {
00177             /* Reference timer is not running, return error. */
00178             cal_result = false;
00179             break;
00180         }
00181 
00182         if (curr_value_cal_timer <= (uint16_t)(TARGETVAL_CALIBRATION))
00183         {
00184             curr_dif_timers = (uint16_t)(TARGETVAL_CALIBRATION) - curr_value_cal_timer;
00185         }
00186         else
00187         {
00188             curr_dif_timers = curr_value_cal_timer - (uint16_t)(TARGETVAL_CALIBRATION);
00189         }
00190 
00191         if (curr_dif_timers < best_dif_timers)
00192         {
00193             best_dif_timers = curr_dif_timers;
00194             best_value_osccal = OSCCAL;
00195         }
00196 
00197         if (curr_value_cal_timer <= (uint16_t)(TARGETVAL_CALIBRATION))
00198         {
00199             /* Too slow, increase speed */
00200             OSCCAL++;
00201         }
00202         else
00203         {
00204             /* Too fast, lower speed */
00205             OSCCAL--;
00206         }
00207     }
00208 
00209     TCCR2B = tccr2b;
00210     TCCR1B = tccr1b;
00211     TCCR1A = tccr1a;
00212     LEAVE_CRITICAL_REGION();
00213 
00214     OSCCAL = best_value_osccal;
00215 
00216     return (cal_result);
00217 }

00296         case LED_ON:

00299             break;
00300 
00301         case LED_OFF:
00302             led_state &= ~_BV(pin);
00303             *pIO_LED = (uint8_t)(~(uint16_t)led_state) | ~0x03;
00304             break;
00305 
00306         case LED_TOGGLE:
00307         default:
00308             if (led_state & _BV(pin))
00309             {
00310                 /*
00311                  * LED is currently on,
00312                  * Switch it off
00313                  */
00314                 led_state &= ~_BV(pin);
00315             }
00316             else
00317             {
00318                 /*
00319                  * LED is currently off,
