00123 uint8_t sio_usb_tx(uint8_t *data, uint8_t length)
00124 {
00125     uint8_t bytes_to_be_written;
00126     uint8_t size;
00127     uint8_t back;
00128 
00129     /*
00130      * USB Tx has been marked as busy, e.g. maybe the terminal is not open.
00131      * In order to continue the execution of the application even if
00132      * USB Tx data cannot be delivered, we are going to return here
00133      * immediately.
00134      * Once USB Tx is not busy anymore, regular USB Tx procedures will
00135      * be resumed.
00136      */
00137     if (usb_tx_busy)
00138     {
00139         return (0);
00140     }
00141 
00142     /*
00143      * Calculate available buffer space
00144      */
00145     if (usb_tx_buf_tail >= usb_tx_buf_head)
00146     {
00147         size = (USB_TX_BUF_SIZE - 1) - (usb_tx_buf_tail - usb_tx_buf_head);
00148     }
00149     else
00150     {
00151         size = (usb_tx_buf_head - 1) - usb_tx_buf_tail;
00152     }
00153 
00154     if (size < length)
00155     {
00156         /* Not enough buffer space available. Use the remaining size */
00157         bytes_to_be_written = size;
00158     }
00159     else
00160     {
00161         bytes_to_be_written = length;
00162     }
00163 
00164     /* Remember the number of bytes transmitted. */
00165     back = bytes_to_be_written;
00166     usb_tx_count += bytes_to_be_written;
00167 
00168     /* The data is copied to the transmit buffer. */
00169     while (bytes_to_be_written > 0)
00170     {
00171         usb_tx_buf[usb_tx_buf_tail] = *data;
00172 
00173         if ((USB_TX_BUF_SIZE - 1) == usb_tx_buf_tail)
00174         {
00175             /* Reached the end of buffer, revert back to beginning of buffer. */
00176             usb_tx_buf_tail = 0;
00177         }
00178         else
00179         {
00180             /*
00181              * Increment the index to point the next character to be
00182              * inserted.
00183              */
00184             usb_tx_buf_tail++;
00185         }
00186 
00187         bytes_to_be_written--;
00188         data++;
00189     }
00190 
00191     /*
00192      * If a transmission needs to be started, pal_usb_handler() takes
00193      * care about that once it is run.
00194      */
00195 
00196     return back;

00239 uint8_t sio_usb_rx(uint8_t *data, uint8_t max_length)
00240 {
00241     uint8_t data_received = 0;
00242 
00243     if (usb_rx_count == 0)
00244     {
00245         /* USB receive buffer is empty. */
00246         return 0;
00247     }
00248 
00249     if (USB_RX_BUF_SIZE <= usb_rx_count)
00250     {
00251          /*
00252           * Bytes between head and tail are overwritten by new data.
00253           * The oldest data in buffer is the one to which the tail is
00254           * pointing. So reading operation should start from the tail.
00255           */
00256         usb_rx_buf_head = usb_rx_buf_tail;
00257 
00258         /*
00259          * This is a buffer overflow case. Byt still only bytes equivalent to
00260          * full buffer size are useful.
00261          */
00262         usb_rx_count = USB_RX_BUF_SIZE;
00263 
00264         /* Bytes received is more than or equal to buffer. */
00265         if (USB_RX_BUF_SIZE <= max_length)
00266         {
00267              /*
00268               * Requested receive length (max_length) is more than the
00269               * max size of receive buffer, but at max the full
00270               * buffer can be read.
00271               */
00272              max_length = USB_RX_BUF_SIZE;
00273         }
00274     }
00275     else
00276     {
00277         /* Bytes received is less than receive buffer maximum length. */
00278         if (max_length > usb_rx_count)
00279         {
00280             /*
00281              * Requested receive length (max_length) is more than the data
00282              * present in receive buffer. Hence only the number of bytes
00283              * present in receive buffer are read.
00284              */
00285             max_length = usb_rx_count;
00286         }
00287     }
00288 
00289     data_received = max_length;
00290 
00291     while (max_length > 0)
00292     {
00293         /* Start to copy from head. */
00294         *data = usb_rx_buf[usb_rx_buf_head];
00295 
00296         if (++usb_rx_buf_head == USB_RX_BUF_SIZE)
00297         {
00298             usb_rx_buf_head = 0;
00299         }
00300 
00301         usb_rx_count--;
00302         data++;
00303         max_length--;
00304     }
00305     return data_received;

00356     if (LOW == (cUSBstat & USB_RXF))
00357     {
00358         /* The received data is read in the PAL USB buffer */
00359         usb_getc();
00360     }
00361 
00362     /* Transmit data, if any */
00363     if (usb_tx_count > 0)
00364     {
00365         uint8_t bytes_tx = MAX_BYTES_FOR_USB_TX;
00366 
00367         while ((usb_tx_count > 0) && (bytes_tx > 0))
00368         {
00369             /*
00370              * To avoid slow transmission (byte per pal_task), a maximum of
00371              * MAX_BYTES_FOR_USB_TX bytes or number of bytes present in USB
00372              * buffer, whichever is minimum is transmitted at one go.

00374             if (LOW == (cUSBstat & USB_TXE))
00375             {
00376                 if (usb_tx_busy)

00379                      * Reset USB busy handling variables and mark USB as not busy.

00383                 }

00386 
00387                 bytes_tx--;
00388             }
00389             else
00390             {
00391                 /*
00392                  * In case USB Tx is already marked busy, we can skip the
00393                  * USB Tx busy detection algorithm.
